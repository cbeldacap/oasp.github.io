11  import javax.annotation.security.RolesAllowed;
12  import javax.inject.Inject;
13  import javax.inject.Named;

15  /**
16   * UseCase: The guests can change a table to get a better table.
17   *
18   * @author rjoeris
19   */
20  @Named
21  public class UcChangeTableImpl extends AbstractUc implements UcChangeTable {
22  
23    private Tablemanagement tableManagement;
24  
25    /**
26     * {@inheritDoc}
27     */
28    @Override
29    @RolesAllowed(PermissionConstants.SAVE_TABLE)
30    public void changeTable(OrderEto order, TableEto newTable) {
31  
32      // save old table data
33      long oldTableId = order.getTableId();
34      TableState oldTableState = this.tableManagement.findTable(oldTableId).getState();
35  
36      // update order
37      order.setTableId(newTable.getId());
38  
39      // TODO
40      // Salesmanagement s;
41      // s.saveOrder(order);
42  
43      // change table status:
44      // marks old table as free
45      this.tableManagement.markTableAs(this.tableManagement.findTable(oldTableId), TableState.FREE);
46      // marks new table with copied status
47      this.tableManagement.markTableAs(newTable, oldTableState);
48  
49    }
50  
51    /**
52     * @param tableManagement the {@link Tablemanagement} to {@link Inject}.
53     */
54    @Inject
55    public void setTableManagement(Tablemanagement tableManagement) {
56  
57      this.tableManagement = tableManagement;
58    }
59  
60  }
