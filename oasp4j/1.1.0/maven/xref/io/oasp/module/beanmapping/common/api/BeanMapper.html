1   package io.oasp.module.beanmapping.common.api;
2   
3   import java.util.List;
4   import java.util.Set;
5   
6   /**
7    * This is the interface used to convert from one Java bean to another compatible bean (e.g. from a JPA entity to a
8    * corresponding transfer-object).
9    * 
10   * @author hohwille
11   */
12  public interface BeanMapper {
13  
14    /**
15     * Recursively converts the given <code>source</code> {@link Object} to the given target {@link Class}.
16     * 
17     * @param <T> is the generic type to convert to.
18     * @param source is the object to convert.
19     * @param targetClass is the {@link Class} reflecting the type to convert to.
20     * @return the converted object. Will be <code>null</code> if <code>source</code> is <code>null</code>.
21     */
22    <T> T map(Object source, Class<T> targetClass);
23  
24    /**
25     * Creates a new {@link List} with the {@link #map(Object, Class) mapped bean} for each {@link List#get(int) entry} of
26     * the given {@link List}.
27     * 
28     * @param <T> is the generic type to convert the {@link List} entries to.
29     * @param source is the {@link List} with the source objects.
30     * @param targetClass is the {@link Class} reflecting the type to convert each {@link List} entry to.
31     * @return the {@link List} with the converted objects. Will be {@link List#isEmpty() empty} is <code>source</code> is
32     *         empty or <code>null</code>.
33     */
34    <T> List<T> mapList(List<?> source, Class<T> targetClass);
35  
36    /**
37     * Creates a new {@link Set} with the {@link #map(Object, Class) mapped bean} for each {@link Set#contains(Object)
38     * entry} of the given {@link Set}.
39     * 
40     * @param <T> is the generic type to convert the {@link Set} entries to.
41     * @param source is the {@link Set} with the source objects.
42     * @param targetClass is the {@link Class} reflecting the type to convert each {@link Set} entry to.
43     * @return the {@link Set} with the converted objects. Will be {@link Set#isEmpty() empty} is <code>source</code> is
44     *         empty or <code>null</code>.
45     */
46    <T> Set<T> mapSet(Set<?> source, Class<T> targetClass);
47  
48  }
