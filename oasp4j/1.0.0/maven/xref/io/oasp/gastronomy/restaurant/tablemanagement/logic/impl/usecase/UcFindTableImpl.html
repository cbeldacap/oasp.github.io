3   import io.oasp.gastronomy.restaurant.general.common.api.constants.PermissionConstants;
4   import io.oasp.gastronomy.restaurant.tablemanagement.dataaccess.api.TableEntity;
5   import io.oasp.gastronomy.restaurant.tablemanagement.logic.api.to.TableEto;

9   import java.util.List;
10  
11  import javax.annotation.security.RolesAllowed;
12  import javax.inject.Named;
13  
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  
17  /**
18   * Implementation of {@link UcFindTable}.
19   *
20   * @author jozitz
21   */
22  @Named
23  public class UcFindTableImpl extends AbstractTableUc implements UcFindTable {
24  
25    /** Logger instance. */
26    private static final Logger LOG = LoggerFactory.getLogger(UcFindTableImpl.class);
27  
28    /**
29     * {@inheritDoc}
30     */
31    @Override
32    @RolesAllowed(PermissionConstants.FIND_TABLE)
33    public TableEto findTable(Long id) {
34  
35      LOG.debug("Get table with id '" + id + "' from database.");
36      return getBeanMapper().map(getTableDao().findOne(id), TableEto.class);
37    }
38  
39    /**
40     * {@inheritDoc}
41     */
42    @Override
43    @RolesAllowed(PermissionConstants.FIND_TABLE)
44    public List<TableEto> findAllTables() {
45  
46      LOG.debug("Get all restaurant tables from database.");
47      List<TableEntity> tables = getTableDao().findAll();
48      return getBeanMapper().mapList(tables, TableEto.class);
49    }
50  
51    /**
52     * {@inheritDoc}
53     */
54    @Override
55    @RolesAllowed(PermissionConstants.FIND_TABLE)
56    public List<TableEto> findFreeTables() {
57  
58      LOG.debug("Get all free restaurant tables from database.");
59  
60      List<TableEntity> tables = getTableDao().getFreeTables();
61      return getBeanMapper().mapList(tables, TableEto.class);
62    }
63  
64  }
