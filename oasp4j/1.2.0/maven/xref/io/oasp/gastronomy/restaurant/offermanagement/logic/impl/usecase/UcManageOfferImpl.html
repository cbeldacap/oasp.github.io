19  /**
20   * Implementation of {@link UcManageOffer}.
21   *
22   * @author loverbec
23   */
24  @Named
25  @UseCase
26  @Validated
27  public class UcManageOfferImpl extends AbstractOfferUc implements UcManageOffer {
28  
29    /**
30     * {@inheritDoc}
31     */
32    @Override
33    @RolesAllowed(PermissionConstants.DELETE_OFFER)
34    public void deleteOffer(Long offerId) {
35  
36      getOfferDao().delete(offerId);
37    }
38  
39    /**
40     * {@inheritDoc}
41     */
42    @Override
43    @RolesAllowed(PermissionConstants.SAVE_OFFER)
44    public OfferEto saveOffer(@Valid OfferEto offer) {
45  
46      Objects.requireNonNull(offer, "offer");
47  
48      if ((offer.getMealId() == null) && (offer.getDrinkId() == null) && (offer.getSideDishId() == null)) {
49        throw new OfferEmptyException(offer);
50      } else {
51        OfferEntity persistedOffer = getOfferDao().save(getBeanMapper().map(offer, OfferEntity.class));
52        return getBeanMapper().map(persistedOffer, OfferEto.class);
53      }
54    }
55  
56  }
