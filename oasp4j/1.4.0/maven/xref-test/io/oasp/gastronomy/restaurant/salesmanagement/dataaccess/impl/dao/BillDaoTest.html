36    private BillDao billDao;
37  
38    @PersistenceContext
39    private EntityManager entityManager;
40  
41    /**
42     * Tests if a Bill is persisted correctly. Special focus is on the mapping of {@link Money} and furthermore the query
43     * of one of the {@link Money} fields is tested.
44     */
45    @Test
46    public void testPersist() {
47  
48      BillEntity bill = new BillEntity();
49      bill.setTotal(new Money(42.42));
50      bill.setTip(new Money(1.0));
51      bill.setPayed(true);
52      assertNull(bill.getId());
53      this.billDao.save(bill);
54      assertNotNull(bill.getId());
55      BillEntity loadedBill = this.billDao.findOne(bill.getId());
56      assertEquals(bill, loadedBill);
57  
58      TypedQuery<BillEntity> query =
59          this.entityManager.createQuery("SELECT b from Bill b where b.total > 43", BillEntity.class);
60      List<BillEntity> resultList = query.getResultList();
61      assertTrue(resultList.isEmpty());
62  
63      query = this.entityManager.createQuery("SELECT b from Bill b where b.total < 43", BillEntity.class);
64      resultList = query.getResultList();
65      assertTrue(!resultList.isEmpty());
66  
67    }
